# PostgreSQL отчёты

## Обзор

PostgreSQL имеет встроенную поддержку для трёх типов отчётов:
- **View** — обычные представления
- **MaterializedView** — материализованные представления (с индексами)
- **Function** — табличные функции с параметрами

## Типы отчётов в PostgreSQL

### 1. View (Обычное представление)

```sql
-- Создание
CREATE VIEW public.report_sales AS
SELECT 
    o.id,
    o.order_date,
    c.customer_name,
    p.product_name,
    od.quantity,
    od.unit_price,
    (od.quantity * od.unit_price) as total_amount
FROM orders o
JOIN customers c ON o.customer_id = c.id
JOIN order_details od ON o.id = od.order_id
JOIN products p ON od.product_id = p.id
WHERE o.order_date >= CURRENT_DATE - INTERVAL '90 days'
ORDER BY o.order_date DESC;

-- Добавить описание
COMMENT ON VIEW public.report_sales IS 'Отчёт о продажах за последние 90 дней';
```

**Особенности:**
- ✅ Автоматически использует индексы
- ✅ Всегда возвращает актуальные данные
- ✅ Нет расхода диска
- ❌ Медленнее материализованных представлений

### 2. MaterializedView (Материализованное представление)

```sql
-- Создание
CREATE MATERIALIZED VIEW public.report_sales_summary AS
SELECT 
    DATE(o.order_date) as sale_date,
    c.region,
    p.category,
    COUNT(DISTINCT o.id) as transaction_count,
    COUNT(DISTINCT c.id) as unique_customers,
    SUM(od.quantity) as total_quantity,
    SUM(od.quantity * od.unit_price) as total_amount,
    AVG(od.quantity * od.unit_price) as avg_transaction
FROM orders o
JOIN customers c ON o.customer_id = c.id
JOIN order_details od ON o.id = od.order_id
JOIN products p ON od.product_id = p.id
WHERE o.order_date >= CURRENT_DATE - INTERVAL '1 year'
GROUP BY DATE(o.order_date), c.region, p.category
ORDER BY sale_date DESC, region, category;

-- Добавить описание
COMMENT ON MATERIALIZED VIEW public.report_sales_summary IS 
'Итоги продаж по дате, регионам и категориям товаров';

-- Создать индекс для ускорения
CREATE INDEX idx_report_sales_summary_date ON public.report_sales_summary(sale_date DESC);
```

**Особенности:**
- ✅ Очень быстрое выполнение
- ✅ Можно создавать индексы
- ✅ Занимает место на диске
- ❌ Требует обновления вручную

**Обновление:**

```sql
-- Стандартное обновление (блокирует доступ)
REFRESH MATERIALIZED VIEW public.report_sales_summary;

-- Параллельное обновление (если есть уникальный индекс)
REFRESH MATERIALIZED VIEW CONCURRENTLY public.report_sales_summary;
```

### 3. Function (Табличная функция)

```sql
-- Создание функции с параметрами
CREATE OR REPLACE FUNCTION public.get_sales_by_period(
    p_start_date DATE DEFAULT CURRENT_DATE - INTERVAL '30 days',
    p_end_date DATE DEFAULT CURRENT_DATE,
    p_region VARCHAR DEFAULT NULL
)
RETURNS TABLE (
    id INTEGER,
    order_date DATE,
    customer_name VARCHAR,
    region VARCHAR,
    total_amount NUMERIC,
    item_count INTEGER
) AS $$
BEGIN
    RETURN QUERY
    SELECT 
        o.id,
        DATE(o.order_date)::DATE,
        c.customer_name,
        c.region,
        SUM(od.quantity * od.unit_price)::NUMERIC,
        COUNT(*)::INTEGER
    FROM orders o
    JOIN customers c ON o.customer_id = c.id
    JOIN order_details od ON o.id = od.order_id
    WHERE DATE(o.order_date) BETWEEN p_start_date AND p_end_date
      AND (p_region IS NULL OR c.region = p_region)
    GROUP BY o.id, DATE(o.order_date), c.customer_name, c.region
    ORDER BY DATE(o.order_date) DESC;
END;
$$ LANGUAGE plpgsql;

-- Добавить описание
COMMENT ON FUNCTION public.get_sales_by_period(DATE, DATE, VARCHAR) IS 
'Получить продажи за период и регион. Параметры: дата начала, дата конца, регион (опционально)';
```

**Особенности:**
- ✅ Параметризованные запросы
- ✅ Сложная бизнес-логика
- ✅ Вычисляемые значения
- ✅ Фильтрация на стороне БД
- ❌ Медленнее материализованных представлений

## Примеры SQL

### Сложная функция с условиями

```sql
CREATE OR REPLACE FUNCTION public.get_top_customers(
    p_min_purchases INTEGER DEFAULT 5,
    p_min_amount NUMERIC DEFAULT 10000,
    p_limit INTEGER DEFAULT 10
)
RETURNS TABLE (
    customer_id INTEGER,
    customer_name VARCHAR,
    purchase_count INTEGER,
    total_spent NUMERIC,
    avg_order_value NUMERIC,
    last_purchase_date DATE
) AS $$
BEGIN
    RETURN QUERY
    SELECT 
        c.id::INTEGER,
        c.customer_name::VARCHAR,
        COUNT(DISTINCT o.id)::INTEGER,
        SUM(od.quantity * od.unit_price)::NUMERIC,
        (SUM(od.quantity * od.unit_price) / COUNT(DISTINCT o.id))::NUMERIC,
        MAX(DATE(o.order_date))::DATE
    FROM customers c
    LEFT JOIN orders o ON c.id = o.customer_id
    LEFT JOIN order_details od ON o.id = od.order_id
    GROUP BY c.id, c.customer_name
    HAVING COUNT(DISTINCT o.id) >= p_min_purchases
       AND COALESCE(SUM(od.quantity * od.unit_price), 0) >= p_min_amount
    ORDER BY SUM(od.quantity * od.unit_price) DESC
    LIMIT p_limit;
END;
$$ LANGUAGE plpgsql;
```

### Функция с динамическим SQL

```sql
CREATE OR REPLACE FUNCTION public.get_report_data(
    p_table_name TEXT,
    p_where_condition TEXT DEFAULT '1=1'
)
RETURNS TABLE (
    col_count INTEGER,
    row_count INTEGER,
    first_row TEXT
) AS $$
DECLARE
    v_count INTEGER;
    v_sample TEXT;
BEGIN
    -- Получить количество колонок
    SELECT COUNT(*)
    INTO col_count
    FROM information_schema.columns
    WHERE table_name = p_table_name;
    
    -- Выполнить запрос с условием
    EXECUTE format(
        'SELECT COUNT(*) FROM %I WHERE %s',
        p_table_name,
        p_where_condition
    ) INTO row_count;
    
    first_row := 'Sample data fetched';
    
    RETURN NEXT;
END;
$$ LANGUAGE plpgsql;
```

## Оптимизация

### Добавление индексов

```sql
-- Для материализованного представления
CREATE INDEX idx_report_sales_summary_date 
ON public.report_sales_summary(sale_date DESC);

CREATE INDEX idx_report_sales_summary_region 
ON public.report_sales_summary(region);

-- Составной индекс для частых фильтров
CREATE INDEX idx_report_sales_summary_compound 
ON public.report_sales_summary(region, sale_date DESC);
```

### Параллельное обновление

```sql
-- Создать уникальный индекс (необходимо для CONCURRENTLY)
CREATE UNIQUE INDEX idx_report_sales_summary_unique 
ON public.report_sales_summary(sale_date, region, category);

-- Теперь можно обновлять параллельно
REFRESH MATERIALIZED VIEW CONCURRENTLY public.report_sales_summary;
```

### Автоматическое обновление

```sql
-- Создать функцию для обновления
CREATE OR REPLACE FUNCTION public.refresh_all_materialized_views()
RETURNS void AS $$
DECLARE
    v_mv_name TEXT;
BEGIN
    FOR v_mv_name IN
        SELECT matviewname FROM pg_matviews
        WHERE schemaname = 'public'
    LOOP
        EXECUTE format('REFRESH MATERIALIZED VIEW %I', v_mv_name);
        RAISE NOTICE 'Updated view: %', v_mv_name;
    END LOOP;
END;
$$ LANGUAGE plpgsql;

-- Запустить вручную
SELECT public.refresh_all_materialized_views();
```

## Интеграция с системой мониторинга

### Система может отслеживать

```sql
-- Размер материализованного представления
SELECT 
    schemaname,
    matviewname,
    pg_size_pretty(pg_total_relation_size(schemaname||'.'||matviewname)) as size
FROM pg_matviews
WHERE schemaname = 'public'
ORDER BY pg_total_relation_size(schemaname||'.'||matviewname) DESC;

-- Время последнего обновления
SELECT 
    schemaname,
    matviewname,
    pg_stat_user_tables.last_vacuum
FROM pg_matviews
JOIN pg_stat_user_tables ON pg_matviews.matviewname = pg_stat_user_tables.relname
WHERE schemaname = 'public';
```

## Обработка ошибок

### Ошибка: Materialized view is not standalone

```
ERROR: cannot refresh materialized view "public.report_sales_summary" without unique index
```

**Решение:** Добавить уникальный индекс

```sql
CREATE UNIQUE INDEX idx_report_unique 
ON public.report_sales_summary(column1, column2);
```

### Ошибка: Function does not return rows

```
ERROR: function get_sales_by_period() does not return rows
```

**Решение:** Использовать SETOF или TABLE

```sql
-- Правильно
CREATE OR REPLACE FUNCTION get_sales(...)
RETURNS TABLE (col1 TYPE, col2 TYPE) AS $$
...
RETURN QUERY SELECT ...
$$ LANGUAGE plpgsql;

-- Или
CREATE OR REPLACE FUNCTION get_sales(...)
RETURNS SETOF record AS $$
...
RETURN NEXT ...
$$ LANGUAGE plpgsql;
```
